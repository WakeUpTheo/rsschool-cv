### JavaScript Code Example from Codewars
**Task description:**  Build a function that returns an array of integers from n to 1 where n>0. Example : n=5 --> [5,4,3,2,1]
```
const reverseSeq = n => {
    let arr = [];
    let i = 0
    do {
        arr.unshift(n - i);
        i++
    } while (i < n)
    return arr.reverse();
};
```
